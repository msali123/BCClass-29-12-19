#### Iterate the process to fetach the links

```python
links = []
for x in range(td):

    #Extracting the article links found after searching the term Selenium
    section = soup.find_all('div', attrs={'class' : 'r'})

    #since the first page is already open we have to extract the links before it moves to the next page
    if(x == 0):
         for elem in section:
                #extracting all the referenced links under a tag
                wrappers = elem.find_all('a', href = True)
                for href_tag in wrappers:
                    link = href_tag['href']
                    links.append({'Article_link':format(link)})


    else:
        #sleeps for 10 sec, to provide time to webscrape the new page opened.
        time.sleep(10)
        #automates clicking next page
        next_page = driver.find_element_by_link_text("Next").click()

        #getting the html code of the new page opened
        current_url = driver.current_url
        driver.get(current_url)

        page_source = driver.page_source
        section_new = BeautifulSoup(page_source, "lxml")
        for elem in section_new:
            wrappers = elem.find_all('a', href = True)
            for href_tag in wrappers:
                link = href_tag['href']
                links.append({"Article_link":format(link)})
```
